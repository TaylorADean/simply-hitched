git status
	- Get the current status

git add
	- Stage a file or add one that hasn't been tracked.

git diff
	- Show the difference between unstaged files and what's in your repo
	
git diff --staged
	- Show the difference between unstaged and staged files

git commit
	- Commit all staged files to local repo

git -a commit
	- Commits all tracked files by skipping staging

git rm
	- Removes a file from GIT and working directory
